public static boolean isRowEmpty(Row row) {

for (int c = row.getFirstCellNum(); c < row.getLastCellNum(); c++) {

Cell cell = row.getCell(c);

if (cell != null && cell.getCellType() != Cell.CELL_TYPE_BLANK)

return false;

}

return true;

}
