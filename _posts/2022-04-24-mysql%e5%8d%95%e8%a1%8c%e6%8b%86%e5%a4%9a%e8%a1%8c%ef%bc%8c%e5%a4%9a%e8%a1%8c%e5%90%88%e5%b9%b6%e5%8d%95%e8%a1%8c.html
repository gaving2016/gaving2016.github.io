分隔效果

-- 分隔前
1,2,3,4
-- 分隔后
1
2
3
4

分隔命令行

SELECT
substring_index(substring_index('1,2,3,4',',', b.help_topic_id + 1), ',', -1) result
FROM
mysql.help_topic b
where
b.help_topic_id < (LENGTH('1,2,3,4') - LENGTH(REPLACE('1,2,3,4', ',', '')) + 1);
